    for stand in range(len(data) - 1): # 2ì¤‘ forë¬¸

    return data

    visited, need_visit = list(), list()
    need_visit.append(start_node)
    
    while need_visit:

            visited.append(node)
            need_visit.extend(graph[node])
    
    return visited
